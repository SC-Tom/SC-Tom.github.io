Basket Schema


Basket {
  costs: {
    subtotal: string;
  };
  currency?: string;
  items: Array;
}

Basket Item Schema


BasketItem {
  costs?: {
    subtotal?: string;
  };
  product: Product;
  quantity: number;
}

Product Schema


export interface Product {
  name: string;
  size: string;
  url: string;
  cost: string;
}
